CREATE PROCEDURE [dbo].[GetInterrailCarrierByCode]

@railCarrierCode char(10)

as 

begin

	select id, RailCarrierCode, RailCarrierName, LastModifiedBy, LastModifiedOn from IRGRailCarrier 
	where RailCarrierCode = @railCarrierCode
	
end
